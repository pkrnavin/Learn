# If id command returns zero, you.ve root access.
if [ $(id -u) -eq 0 ];

  PS1="[\\u@$(tput setaf 1)\\]test\\[$(tput sgr0)\\]@\\h: \\W]# "
else # normal
  PS1="[\\u@\\h:\\w] $"
fi

--VI editor tips:
yy - to copy a line
p - to paste the copied line
6p - to paste the copied line 6 times, as 6 rows
dd - deletea  line
6dd - delete 6 lines
/text - to search a text
n - to find next search occurance
N - to find previous occurance of search word
?text - to find previous occurance of search word
ctrl+g - current line no.
0 - go to first char of the line
$ - go to last char of the line
A - go to one char after last char. & start typing
:0 - go to last char of the file
:$ - go to last char of the file
:23 - go to 23rd line
ctrl+u - pgup
ctrl+d - pgdn
ctrl+f - scroll down 1 full page(page - screen hieght)
ctrl+b - scroll up 1 full page(page - screen hieght)
w - next word
W - next line
b - previous word
u - undo
